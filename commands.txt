For docker
===========
# Launching the server
docker-compose build
docker-compose up


For postgres
===========
# Launching postgres container
docker exec -it postgres /bin/bash 

# Accessing the postgres environment from the postgres role name
psql -u postgres 

# Listing all databases (Don't forget the \g at the end of every sql syntax to execute it)
SELECT datname FROM pg_database

# Alter password for a particular user role name
alter postgres 'rolename' with encrypted password 'password'

# Create a new database
CREATE DATABASE 'database'

# Migrate
python3 manage.py migrate
